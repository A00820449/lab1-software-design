abstract class PizzaStore {
  {abstract} CheesePizza createCheesePizza()
  {abstract} VeganPizza createVeganPizza()
  {abstract} HawaiianPizza createHawaiianPizza()
  
  Pizza orderPizza(String request)
  
}
